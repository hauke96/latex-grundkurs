% Defines an environment for latexcode
\newenvironment{latexcode}
 {\VerbatimEnvironment
  \begin{VerbatimOut}{latexcode.out}}
  {\end{VerbatimOut}%
  \inputminted[tabsize=4]{latex}{latexcode.out}}

\newenvironment{smalllatexcode}
 {\VerbatimEnvironment
  \begin{VerbatimOut}{latexcode.out}}
  {\end{VerbatimOut}%
  \inputminted[fontsize=\scriptsize,tabsize=4]{latex}{latexcode.out}}

\newenvironment{smallerlatexcode}
 {\VerbatimEnvironment
  \begin{VerbatimOut}{latexcode.out}}
  {\end{VerbatimOut}%
  \inputminted[fontsize=\small,tabsize=4]{latex}{latexcode.out}}
